### Исходный код
```cpp

#include <iostream>
using namespace std;
int const N = 3;
int S[N]{1,2,3}; 

void backtrack(int n){
    if (n == N) {
        cout << S[0] << " " << S[1] << " " << S[2] << endl;
        return;
    }

    for (int i = n; i < N; i++){
        int temp = S[n];
        S[n] = S[i];
        S[i] = temp;

        backtrack(S, n + 1);

        temp = S[n];
        S[n] = S[i];
        S[i] = temp;
    }
} 

int main() {  
    backtrack(0);
```
